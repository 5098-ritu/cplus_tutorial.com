                  String(char ch[]){
                     int i=0;
                     while(ch[i]!='\0'){
                        s[i]=ch[i];
                        i++;
                     }
                     s[i]='\0';
                  }
                  String(){
                    s[0]='\0';
                  }
                  String operator +(String &st){
                    String temp;
                    int i=0;
                     while(s[i]!='\0'){
                        temp.s[i]=s[i];
                        i++;
                     }
                     int j=0;
                     while(st.s[j]!='\0'){
                        temp.s[i]=st.s[j];
                        i++;
                        j++;
                     }
                     temp.s[i]='\0';
                     return temp;
                  }
                  void display(){
                    cout<< s<< endl;
                  }
                };
