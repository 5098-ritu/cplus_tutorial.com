            #include< iostream.h>
                
                int main(){
                  int ar[]={1,2,6,4,3,5},l,grt;
                  l=sizeof(ar)/sizeof(int);
                  grt=ar[0];
                  for(int i=0;i< l;i++){
                    if(grt< ar[i]){
                      grt=ar[i];
                  }}
                  cout<<"Array : ";
                  for(int i=0;i< l;i++){
                     cout<< ar[i]<<"  ";
                  }
