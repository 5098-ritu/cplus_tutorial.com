Create a class showing an example of parameterized constructor:

            #include < iostream.h>
                #include < string.h>
                
                
                class a {
                private:
                    int i;
                public:
                    a(int num) {
                        i = num;
                    }
                    void display() {
                        cout << "Value of i: " << i << " using parameterized constructor" << endl;
                    }
                };
                
                int main() {
                    a obj(4);
                    obj.display();
                    return 0;
                }

Output:

            Value of i: 4 using parameterized constructor
