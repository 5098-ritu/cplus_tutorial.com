Create a class entering the rollno, name and class of the student from user but rollno should be automatically generated as we enter the information of 10 students:

            #include < iostream.h>
                #include < string.h>
                
                class Student {
                private:
                    static int rollCounter;
                    int rn; 
                    string nm; 
                    string cl;
                
                public:
                    Student() {
                        rn = ++rollCounter;
                    }
                
                    Student(string name, string className) : nm(name), cl(className) {
                        rn = ++rollCounter;
                    }
                
                    void display() {
                        cout << "Roll No: " << rn << ", Name: " << nm << ", Class: " << cl << endl;
                    }
                };
                
                int Student::rollCounter = 0;
                
                int main() {
                    Student students[10];
                    string name, cName;
                
                    for (int i = 0; i < 10; ++i) {
                        cout << "Enter name of student " << i + 1 << ": ";
                        getline(cin, name);
                        cout << "Enter class of student " << i + 1 << ": ";
                        getline(cin, cName);
                        students[i] = Student(name, cName);
                    }
                    cout << "Students Information:" << endl;
                    for (int i = 0; i < 10; ++i) {
                        students[i].display();
                    }
                
                    return 0;
                }
