Define a class Date to represent dates. Overload the < operator to compare two dates:

            #include< iostream.h>
                class Date{
                  int month;
                  int day;
                  int year;
                  public:
                  Date(int a,int b,int c){
                    day=a;
                    month=b;
                    year=c;
                    simp();
                  }
                  int operator <(Date &d){
                     if(year< d.year){
                       return true;
                     }
                     else{
                      return false;
                     }
                     if(month< d.month){
                        return true;
                     }
                     else{
                      return false;
                     }
                     if(day< d.day){
                      return true;
                     }
                     else{
                      return false;
                     }
                  }
                  void display(){
                    cout<< day<<"/"<< month<<"/"<< year<< endl;
                  }
                  void simp(){
                    day=day%31;
                    month=month+(day/12);
                    year=year+(month/12);
                    month=month%12;
                  }
                };
