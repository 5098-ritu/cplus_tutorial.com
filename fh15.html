Write a C++ program to calculate the average of numbers stored in a file:

            #include < iostream.h >
#include < fstream.h >
#include < vector.h >
#include < numeric.h >
#include < iomanip.h >

double calculateAverage(const std::string& filename) {
    // Open the file for reading
    std::ifstream file(filename);

    // Check if the file is opened successfully
    if (!file) {
        std::cerr << "Failed to open the file: " << filename << std::endl;
        return 0.0;
    }

    // Read numbers from the file into a vector
    std::vector numbers;
    double number;
    while (file >> number) {
        numbers.push_back(number);
    }

    // Close the file
    file.close();

    // Calculate the sum of numbers
    double sum = std::accumulate(numbers.begin(), numbers.end(), 0.0);

    // Calculate the average
    double average = sum / numbers.size();

    return average;
}

int main() {
    // Specify the filename containing numbers
    std::string filename = "numbers.txt";

    // Calculate the average of numbers stored in the file
    double avg = calculateAverage(filename);

    // Display the average with 2 decimal places
    std::cout << "Average of numbers in the file: " << std::fixed << std::setprecision(2) << avg << std::endl;

    return 0;
}
