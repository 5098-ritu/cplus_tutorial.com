Program to find all subsets of a string:

            #include < iostream.h>
                #include < string.h>
                #include < vector.h>
                
                void generate_subsets(const std::string& str, int index, std::string current_subset, std::vector& subsets) {
                    // Add the current subset to the list of subsets
                    subsets.push_back(current_subset);
                
                    // Recursively generate subsets by including and excluding the current character
                    for (int i = index; i < str.length(); ++i) {
                        generate_subsets(str, i + 1, current_subset + str[i], subsets); // Include the current character
                        generate_subsets(str, i + 1, current_subset, subsets);          // Exclude the current character
                    }
                }
                
                int main() {
                    std::string input_string;
                
                    // Input string
                    std::cout << "Enter a string: ";
                    std::cin >> input_string;
                
                    // Vector to store all subsets
                    std::vector< std::string> subsets;
                
                    // Generate subsets
                    generate_subsets(input_string, 0, "", subsets);
                
                    // Output result
                    std::cout << "All subsets of the string:" << std::endl;
                    for (const auto& subset : subsets) {
                        std::cout << subset << std::endl;
                    }
                
                    return 0;
                }
