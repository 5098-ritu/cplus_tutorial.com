Write a C++ program to create a class called BankAccount with private instance variables accountNumber and balance. Provide public getter and setter methods to access and modify these variable:

            #include < iostream.h>
                #include < string.h>
            
                
                class BankAccount {
                private:
                    string accNo;
                    double bal;
                
                public:
                    void setter(string accNumber, double balance) {
                        accNo = accNumber;
                        bal = balance;
                    }
                
                    void getter() const {
                        cout << "Account Number: " << accNo << ", Balance: Rs." << bal << endl;
                    }
                };
                
                int main() {
                    BankAccount account;
                    account.setter("AC123456789", 7000.0);
                    cout << "Account Information: ";
                    account.getter();
                    return 0;
                }

Output:

            Account Information: Account Number: AC123456789, Balance: Rs.7000
