Write a C++ program to create a class called "Student" with a name, grade, and courses attributes, and methods to add and remove courses:

            #include< iostream.h>
                #include< string.h>
    
                class course{
                   string cname;
                   public:
                   void setc(){
                     cin>>cname;
                   }
                   string proc(){
                     return cname;
                   }
                };
                class Student{
                string name;
                char grade;
                course c[10];
                int cno;
                public:
                Student(){
                  cno=0;
                }
                void get(){
                  cout<<"Name of the student : ";

                  }
                void add(){
                  cout<<"Name of the course : ";
                  c[cno].setc();
                  cno++;
                }
                int find(string ch){
                   for(int i=0;i< cno;i++){
                    if (c[i].proc()==ch){
                      return  i;
                    }
                   }
                   return -1;
                }
                void remove(string ch){
                  int i=find(ch);
                  if(i>=0){
                    cout<< ch<<" course is removed.";
                     for(int j=i;j< cno;j++){
                        c[j]=c[j+1];
                     }
                     cno--;
                  }
                  else{
                    cout<<"No course found with such name. ";
                  }
                }
                };
