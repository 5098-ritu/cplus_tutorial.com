C++ Program to convert Hex to Decimal:

            #include< iostream.h>
                #include< conio.h>
                #include< string.h>
                #include< math.h>
                
                int main()
                {
                    char num[20];
                    int i,r,len,hex=0;
                    clrscr();
                    cout<<"Enter a hexadecimal number : ";
                    cin>>num;
                    len=strlen(num);
                    for(i=0;num[i]!='\0';i++)
                    {
                        len--;
                        if(num[i]>='0'&&num[i]<='9')
                            r=num[i]-48;
                        else if(num[i]>='a'&&num[i]<='f')
                            r=num[i]-87;
                        else if(num[i]>='A'&&num[i]<='F')
                            r=num[i]-55;
                        hex+=r*pow(16,len);
                    }
                    cout<<"\nDecimal equivalent of "<< num<<" is : "<< hex;
                    getch();
                    return 0;
                }
