Write a C++ program to create an abstract class Instrument with abstract methods play() and tune(). Create derived classes for Glockenspiel and Violin that extend the Instrument class and implement the respective methods to play and tune each instrument:

            #include < iostream.h >
#include < conio.h >
class Instrument {
public:
    virtual void play() = 0;
    virtual void tune() = 0;
};

class Glockenspiel : public Instrument {
public:
    void play() override {
        std::cout << "Playing the glockenspiel: ding ding ding!" << std::endl;
    }

    void tune() override {
        std::cout << "Tuning the glockenspiel: adjusting the metal bars." << std::endl;
    }
};

class Violin : public Instrument {
public:
    void play() override {
        std::cout << "Playing the violin: screech screech screech!" << std::endl;
    }

    void tune() override {
        std::cout << "Tuning the violin: adjusting the strings and pegs." << std::endl;
    }
};

int main() {
    Glockenspiel glockenspiel;
    Violin violin;

    std::cout << "Playing and tuning the Glockenspiel:" << std::endl;
    glockenspiel.play();
    glockenspiel.tune();

    std::cout << "\nPlaying and tuning the Violin:" << std::endl;
    violin.play();
    violin.tune();

    return 0;
}
