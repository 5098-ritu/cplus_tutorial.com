Create a class showing an example of default constructor:

            #include < iostream.h>
                
                class a {
                private:
                    int i;
                public:
                    a() {
                        i = 0;
                    }
                    void display() {
                        cout << "Value of i: " << i << " using default constructor" << endl;
                    }
                };
                
                int main() {
                    a obj;
                    obj.display();
                    return 0;
                }

Output:

            Value of i: 0 using default constructor
