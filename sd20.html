C++ Program Find Smallest Number in an array:

            #include< iostream.h>
            
                int main(){
                  int ar[]={3,8,57,86,55,76,56};
                  int sml=ar[0];
                  cout<<"Array : "<< endl;
                  int l=sizeof(ar)/sizeof(int);
                  for(int i=0;i< l;i++){
                    cout<< ar[i]<<" ";
                  }
                  for(int i=0;i< l;i++){
                    if(sml>ar[i]){
                      sml=ar[i];}
                  }
                  cout<< endl<<"Smallest in an array : "<< sml; 
                  return 0;}
