Write a C++ program to create a class called "Airplane" with a flight number, destination, and departure time attributes, and methods to check flight status and delay:

            #include< iostream.h>
                #include< string.h>
        
                class Airplane{
                   int fno;

                   int dh,dm;

                    dm=0;
                   }

                    cout<<"Enter flight no . : ";

                        cout<<"Flight is delayed and will arrive at "<< dh<<":"<< dm<<" to "<< destination<< endl;
                      }
                      else{
                        cout<<"Flight will arrive at "<< dh<<":"<< dm<<" to "<< destination<< endl;
                      }
                   }
                   void fdelay(){
                    cout<<"Enter the new departure time in (HH MM) : ";
                    cin>>dh>>dm;
